-- =====================================================
-- 新逸公寓租金管理系统 V2.0.5
-- SQLite 数据库脚本（17张表）
--
-- Phase 4 新增：
-- - users 表：用户账号（admin/admin123）
-- - permissions 表：角色权限配置
--
-- Phase 3 新增：
-- - meter_readings 表（抄表记录，与账单生成解耦）
-- - monthly_summary_cache 表（月度汇总缓存）
--
-- Phase 2 新增：
-- - 账单操作：确认支付/部分支付/作废（BillStatus 状态机）
--
-- 测试覆盖：89 个集成测试全绿
-- =====================================================

-- 初始化外键约束
PRAGMA foreign_keys = ON;

-- =====================================================
-- 表1: 房间表 (rooms)
-- =====================================================
CREATE TABLE IF NOT EXISTS rooms (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    room_number TEXT NOT NULL UNIQUE,              -- '101', '1203' 等
    floor INTEGER,                                 -- 楼层 1-14
    building TEXT DEFAULT '58栋',                  -- 楼栋

    -- 租金配置（单位：分）
    base_rent INTEGER DEFAULT 0,                   -- 标准月租金
    property_fee INTEGER DEFAULT 0,                -- 每月物业费
    deposit INTEGER DEFAULT 0,                    -- 标准押金

    -- 状态（只保留当前状态，历史查 room_status_log）
    status TEXT DEFAULT '空房'
        CHECK(status IN ('空房','在租','员工','管理','违约','维修中','待清洁')),

    -- 当前水电表读数（V2.0.5 后：仅在无 meter_readings 时作为兜底数据源）
    water_meter_current INTEGER DEFAULT 0,         -- 当前水表读数
    electric_meter_current INTEGER DEFAULT 0,     -- 当前电表读数

    -- 软删除
    is_deleted INTEGER DEFAULT 0,                  -- 0=正常 1=已删除
    deleted_at TEXT,
    deleted_by TEXT,

    created_at TEXT DEFAULT (datetime('now')),
    updated_at TEXT
);

CREATE INDEX IF NOT EXISTS idx_rooms_number ON rooms(room_number);
CREATE INDEX IF NOT EXISTS idx_rooms_floor ON rooms(floor);
CREATE INDEX IF NOT EXISTS idx_rooms_status ON rooms(status);
CREATE INDEX IF NOT EXISTS idx_rooms_deleted ON rooms(is_deleted);

-- =====================================================
-- 表2: 租客表 (tenants)
-- =====================================================
CREATE TABLE IF NOT EXISTS tenants (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,                           -- 姓名（必填）
    phone TEXT NOT NULL,                          -- 手机号（必填）
    phone2 TEXT,                                  -- 备用手机号

    emergency_contact TEXT,                       -- 紧急联系人
    emergency_phone TEXT,                          -- 紧急联系电话

    -- 唯一性说明：手机号唯一性由应用层控制
    -- 原因：运营商回收号码后可能重新分配，不应仅凭手机号判断同一租客
    -- 查询同一人时，通过 tenant_history 表追踪手机号变更历史

    -- 软删除
    is_deleted INTEGER DEFAULT 0,
    deleted_at TEXT,
    deleted_by TEXT,

    created_at TEXT DEFAULT (datetime('now')),
    updated_at TEXT
);

CREATE INDEX IF NOT EXISTS idx_tenants_name ON tenants(name);
CREATE INDEX IF NOT EXISTS idx_tenants_phone ON tenants(phone);
CREATE INDEX IF NOT EXISTS idx_tenants_deleted ON tenants(is_deleted);

-- =====================================================
-- 表3: 合同表 (leases)
-- =====================================================
CREATE TABLE IF NOT EXISTS leases (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    room_id INTEGER NOT NULL REFERENCES rooms(id),
    tenant_id INTEGER NOT NULL REFERENCES tenants(id),

    -- 合同信息（单位：分）
    contract_number TEXT,                        -- 合同编号
    start_date TEXT NOT NULL,                    -- 合同开始日期（格式：YYYY-MM-DD）
    end_date TEXT,                              -- 合同结束日期
    monthly_rent INTEGER DEFAULT 0,              -- 月租金（可优惠）
    property_fee INTEGER DEFAULT 0,              -- 每月物业费
    deposit INTEGER DEFAULT 0,                   -- 押金

    -- 押金管理（单位：分）
    deposit_received INTEGER DEFAULT 0,          -- 已收押金金额
    deposit_balance INTEGER DEFAULT 0,            -- 押金余额
    deposit_status TEXT DEFAULT '未收取'
        CHECK(deposit_status IN ('未收取','已收取','部分收取','退还','没收')),

    -- 实际入住/退房
    move_in_date TEXT,                           -- 实际入住日期
    move_out_date TEXT,                          -- 实际退房日期
    termination_reason TEXT,                    -- 退房原因：正常退房/违约退房/换房/其他

    -- 状态（状态机由应用层控制）
    -- 草稿 → 生效中 → 待结算 → 已退房 → 已归档
    --                ↘ 违约中 → 待结算
    status TEXT DEFAULT '草稿'
        CHECK(status IN ('草稿','生效中','违约中','待结算','已退房','已作废')),
    status_reason TEXT,                          -- 状态变更原因

    -- 软删除
    is_deleted INTEGER DEFAULT 0,
    deleted_at TEXT,
    deleted_by TEXT,

    notes TEXT,

    created_at TEXT DEFAULT (datetime('now')),
    updated_at TEXT
);

CREATE INDEX IF NOT EXISTS idx_leases_room ON leases(room_id);
CREATE INDEX IF NOT EXISTS idx_leases_tenant ON leases(tenant_id);
CREATE INDEX IF NOT EXISTS idx_leases_status ON leases(status);
CREATE INDEX IF NOT EXISTS idx_leases_dates ON leases(start_date, end_date);
CREATE INDEX IF NOT EXISTS idx_leases_deleted ON leases(is_deleted);

-- 复合索引优化
CREATE INDEX IF NOT EXISTS idx_leases_tenant_status ON leases(tenant_id, status);

-- =====================================================
-- 表2.1: 租客历史表 (tenant_history)
-- 注意：放在leases之后，因为引用了leases(id)
-- =====================================================
CREATE TABLE IF NOT EXISTS tenant_history (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    tenant_id INTEGER NOT NULL REFERENCES tenants(id),

    event_type TEXT NOT NULL
        CHECK(event_type IN ('新入住','换房','换号','退房','恢复')),

    -- 关联信息
    room_id INTEGER REFERENCES rooms(id),
    lease_id INTEGER REFERENCES leases(id),

    -- 变更详情
    event_date TEXT NOT NULL,                   -- 事件日期（格式：YYYY-MM-DD）

    old_value TEXT,                              -- 变更前的值（如旧手机号、旧房间号）
    new_value TEXT,                             -- 变更后的值（如新手机号、新房间号）

    notes TEXT,                                 -- 备注

    created_at TEXT DEFAULT (datetime('now'))
);

CREATE INDEX IF NOT EXISTS idx_tenant_history_tenant ON tenant_history(tenant_id);
CREATE INDEX IF NOT EXISTS idx_tenant_history_room ON tenant_history(room_id);
CREATE INDEX IF NOT EXISTS idx_tenant_history_date ON tenant_history(event_date);

-- =====================================================
-- 表4: 月度账单表 (monthly_bills)
-- =====================================================
CREATE TABLE IF NOT EXISTS monthly_bills (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    year_month TEXT NOT NULL,                    -- '2026-03'（格式：YYYY-MM）
    room_id INTEGER NOT NULL REFERENCES rooms(id),
    lease_id INTEGER REFERENCES leases(id),

    -- 入住/退房天数计算
    lease_start_date TEXT,                       -- 合同开始日期（用于计算）
    lease_end_date TEXT,                        -- 合同结束日期（用于计算）
    check_in_day INTEGER,                        -- 入住日（月中入住时）
    check_out_day INTEGER,                       -- 退房日（月中退房时）

    -- 抄表数据（V2.0.5 后：来自 meter_readings 表，不再是 rooms.meter_current）
    water_reading_prev INTEGER DEFAULT 0,        -- 上月水表读数
    water_reading_current INTEGER,               -- 本月水表读数
    electric_reading_prev INTEGER DEFAULT 0,   -- 上月电表读数
    electric_reading_current INTEGER,            -- 本月电表读数

    -- 用量（由应用层计算后写入）
    water_usage INTEGER DEFAULT 0,               -- 用水量
    electric_usage INTEGER DEFAULT 0,            -- 用电量

    -- 单价快照（生成账单时记录，防止后续调价影响历史）
    water_unit_price INTEGER DEFAULT 600,        -- 水费单价（分/方），默认6元=600分
    electric_unit_price INTEGER DEFAULT 73,       -- 电费单价（分/度），默认0.73元=73分
    management_unit_price INTEGER DEFAULT 57,    -- 管理费单价（分/度），默认0.57元=57分

    -- 费用明细（单位：分，由应用层计算）
    rent_fee INTEGER DEFAULT 0,                   -- 房租
    rent_days INTEGER DEFAULT 0,                 -- 租金天数（0=全月，>0=实际天数）
    rent_daily_rate INTEGER DEFAULT 0,           -- 日租金（分）

    property_fee INTEGER DEFAULT 0,              -- 物业费
    water_fee INTEGER DEFAULT 0,                 -- 水费
    electric_fee INTEGER DEFAULT 0,              -- 电费
    management_fee INTEGER DEFAULT 0,            -- 管理费
    repair_fee INTEGER DEFAULT 0,                -- 维修费
    deposit_fee INTEGER DEFAULT 0,               -- 押金（仅新租首月收取）

    -- 欠费结转（单位：分）
    previous_balance INTEGER DEFAULT 0,          -- 上期欠费
    actual_paid INTEGER DEFAULT 0,              -- 当期实际缴费

    -- 合计（单位：分）
    total_amount INTEGER DEFAULT 0,              -- 应缴总额 = 上期欠费 + 当期费用 - 实际缴费

    -- 账单类型
    bill_type TEXT DEFAULT '正常'
        CHECK(bill_type IN ('正常','首月免水电','半月结算','末月结算','月中退房结算')),
    room_status TEXT NOT NULL
        CHECK(room_status IN ('在租','新租','员工','管理','违约','空房','退房')),

    -- 状态（V2.0.5 新增状态机流转规则，见应用层 BillStatus）
    -- 待缴费 → [确认支付] → 已支付
    -- 待缴费 → [部分支付] → 部分支付（可再付至完成）
    -- 待缴费 → [作废] → 已作废 → [重新生成] → 待缴费（UPDATE 恢复，非 INSERT）
    -- 部分支付 → [确认支付] → 已支付
    -- 部分支付 → [作废] → 已作废
    status TEXT DEFAULT '待缴费'
        CHECK(status IN ('待缴费','已支付','部分支付','已作废')),
    due_date TEXT,                              -- 应缴日期
    paid_date TEXT,                            -- 实缴日期

    -- 账单序号（同一房间同一月可能有多张账单，如月中退房+新租）
    bill_sequence INTEGER DEFAULT 1,             -- 第几张账单

    -- 软删除
    is_deleted INTEGER DEFAULT 0,
    deleted_at TEXT,
    deleted_by TEXT,

    -- 备注
    notes TEXT,

    created_at TEXT DEFAULT (datetime('now')),
    updated_at TEXT,

    UNIQUE(year_month, room_id, bill_sequence)
);

CREATE INDEX IF NOT EXISTS idx_bills_yearmonth ON monthly_bills(year_month);
CREATE INDEX IF NOT EXISTS idx_bills_room ON monthly_bills(room_id);
CREATE INDEX IF NOT EXISTS idx_bills_status ON monthly_bills(status);
CREATE INDEX IF NOT EXISTS idx_bills_type ON monthly_bills(bill_type);
CREATE INDEX IF NOT EXISTS idx_bills_deleted ON monthly_bills(is_deleted);

-- 复合索引优化
CREATE INDEX IF NOT EXISTS idx_bills_room_month ON monthly_bills(room_id, year_month);

-- =====================================================
-- 表5: 缴费记录表 (payments)
-- =====================================================
CREATE TABLE IF NOT EXISTS payments (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    bill_id INTEGER REFERENCES monthly_bills(id),
    room_id INTEGER NOT NULL REFERENCES rooms(id),
    lease_id INTEGER REFERENCES leases(id),

    -- 金额（单位：分）
    amount INTEGER NOT NULL,                     -- 缴费金额
    payment_date TEXT,                           -- 缴费日期（格式：YYYY-MM-DD）
    payment_method TEXT
        CHECK(payment_method IN ('微信','支付宝','银行卡','现金','商家码','押金抵扣','混合支付')),

    -- 混合支付时拆分（单位：分）
    wechat_amount INTEGER DEFAULT 0,             -- 微信金额
    alipay_amount INTEGER DEFAULT 0,            -- 支付宝金额
    cash_amount INTEGER DEFAULT 0,              -- 现金金额
    bank_amount INTEGER DEFAULT 0,               -- 银行卡金额
    deposit_deduct_amount INTEGER DEFAULT 0,   -- 押金抵扣金额

    payer_name TEXT,                            -- 付款人
    confirmation_screenshot TEXT,                -- 确认截图路径

    operator TEXT,                              -- 经办人

    -- 软删除
    is_deleted INTEGER DEFAULT 0,
    deleted_at TEXT,
    deleted_by TEXT,

    notes TEXT,

    created_at TEXT DEFAULT (datetime('now'))
);

CREATE INDEX IF NOT EXISTS idx_payments_bill ON payments(bill_id);
CREATE INDEX IF NOT EXISTS idx_payments_room ON payments(room_id);
CREATE INDEX IF NOT EXISTS idx_payments_date ON payments(payment_date);
CREATE INDEX IF NOT EXISTS idx_payments_method ON payments(payment_method);
CREATE INDEX IF NOT EXISTS idx_payments_deleted ON payments(is_deleted);

-- 复合索引优化
CREATE INDEX IF NOT EXISTS idx_payments_room_date ON payments(room_id, payment_date);

-- =====================================================
-- 表14: 抄表记录表 (meter_readings)          [V2.0.5 新增]
-- =====================================================
--
-- V2.0.5 核心变更：抄表录入与账单生成解耦
--
-- 旧数据流（V2.0.3 及之前）:
--   generate_room_bill → 读 rooms.water_meter_current / electric_meter_current → INSERT monthly_bills
--
-- 新数据流（V2.0.5）:
--   抄表录入 → INSERT meter_readings（幂等：同一房间同年月的记录 UNIQUE 约束）
--   generate_room_bill → 读 meter_readings → INSERT monthly_bills
--
-- 倒拨校验：每条记录必须 >= 该房间上一条记录的读数，否则拒绝
-- 换表标记：is_replacement=1 时跳过倒拨校验（换新表从0开始）
--
CREATE TABLE IF NOT EXISTS meter_readings (
    id INTEGER PRIMARY KEY AUTOINCREMENT,

    -- 基本信息

-- 视图1: 房间当前状态视图

CREATE VIEW IF NOT EXISTS v_room_current_status AS
SELECT
    r.id,
    r.room_number,
    r.floor,
    r.status,
    r.base_rent / 100.0 AS base_rent_yuan,           -- 转换为元
    r.property_fee / 100.0 AS property_fee_yuan,
    r.deposit / 100.0 AS deposit_yuan,
    r.water_meter_current,
    r.electric_meter_current,
    t.id AS tenant_id,
    t.name AS tenant_name,
    t.phone AS tenant_phone,
    l.id AS lease_id,
    l.monthly_rent / 100.0 AS monthly_rent_yuan,
    l.start_date AS lease_start,
    l.end_date AS lease_end,
    l.status AS lease_status,
    l.deposit_balance / 100.0 AS deposit_balance_yuan
FROM rooms r
LEFT JOIN leases l ON l.room_id = r.id AND l.status = '生效中' AND l.is_deleted = 0
LEFT JOIN tenants t ON t.id = l.tenant_id AND t.is_deleted = 0
WHERE r.is_deleted = 0;

-- =====================================================
-- 视图2: 月度汇总视图（用于参考，实际使用 monthly_summary_cache）
-- =====================================================
CREATE VIEW IF NOT EXISTS v_monthly_summary AS
SELECT
    year_month,
    COUNT(*) AS room_count,
    SUM(CASE WHEN room_status = '在租' THEN 1 ELSE 0 END) AS rented_count,
    SUM(CASE WHEN room_status = '新租' THEN 1 ELSE 0 END) AS new_rented_count,
    SUM(CASE WHEN room_status = '员工' THEN 1 ELSE 0 END) AS staff_count,
    SUM(CASE WHEN room_status = '空房' THEN 1 ELSE 0 END) AS vacant_count,
    SUM(CASE WHEN room_status = '违约' THEN 1 ELSE 0 END) AS violation_count,
    SUM(CASE WHEN room_status = '退房' THEN 1 ELSE 0 END) AS checkout_count,
    SUM(CASE WHEN room_status = '管理' THEN 1 ELSE 0 END) AS management_count,
    SUM(rent_fee) / 100.0 AS rent_total_yuan,
    SUM(property_fee) / 100.0 AS property_total_yuan,
    SUM(water_fee) / 100.0 AS water_total_yuan,
    SUM(electric_fee) / 100.0 AS electric_total_yuan,
    SUM(management_fee) / 100.0 AS management_total_yuan,
    SUM(repair_fee) / 100.0 AS repair_total_yuan,
    SUM(deposit_fee) / 100.0 AS deposit_total_yuan,
    SUM(previous_balance) / 100.0 AS previous_balance_yuan,
    SUM(actual_paid) / 100.0 AS actual_paid_yuan,
    SUM(total_amount) / 100.0 AS grand_total_yuan
FROM monthly_bills
WHERE is_deleted = 0
GROUP BY year_month;

-- 视图4: 租客消费历史视图
-- =====================================================
CREATE VIEW IF NOT EXISTS v_tenant_consumption AS
SELECT
    l.id AS lease_id,
    r.room_number,
    t.name AS tenant_name,
    t.phone,
    l.start_date AS check_in_date,
    l.move_out_date AS check_out_date,
    l.monthly_rent / 100.0 AS monthly_rent_yuan,
    SUM(b.rent_fee) / 100.0 AS total_rent_yuan,
    SUM(b.water_fee + b.electric_fee + b.management_fee) / 100.0 AS total_utilities_yuan,
    SUM(b.repair_fee) / 100.0 AS total_repair_yuan,
    SUM(b.total_amount) / 100.0 AS total_billed_yuan,
    COUNT(b.id) AS months_count
FROM leases l
JOIN rooms r ON r.id = l.room_id
JOIN tenants t ON t.id = l.tenant_id
LEFT JOIN monthly_bills b ON b.lease_id = l.id AND b.is_deleted = 0
WHERE l.is_deleted = 0
GROUP BY l.id;

-- V2.0.5 TOCTOU 防护规范（queries.rs 顶部注释）
-- =====================================================
--
-- 所有 Service 层的写前读取必须使用 _tx 版本，禁止在事务外读取后事务内写入。
--
-- 已知违规模式（❌）：
--   let conn = ctx.get_conn()?;
--   let room = get_room_by_id(&conn, ...)?;    // 事务外读取
--   ctx.transaction(|tx| { tx.execute(...) })?; // 事务内写入
--
-- 正确模式（✅）：
--   ctx.transaction(|tx| {
--       let room = get_room_by_id_tx(tx, ...)?; // 事务内读取
--       tx.execute(...);                          // 事务内写入
--   })?;
--
-- 新增查询接口时，必须同时提供 &Connection 和 &Transaction 两个版本。
--
-- V2.0.5 新增 TOCTOU 防护范围：
-- - meter_reading_service: record_meter_reading（倒拨校验在事务内）
-- - bill_service: confirm_bill_paid / partial_pay_bill / void_bill
