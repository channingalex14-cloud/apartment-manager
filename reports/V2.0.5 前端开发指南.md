## 二、Tauri API 调用规范

### 2.1 invoke 封装

所有后端命令通过 `invoke()` 调用，错误统一处理：

```typescript
// src/services/api.ts
import { invoke } from "@tauri-apps/api/core";

export async function callCommand<T>(cmd: string, args?: Record<string, unknown>): Promise<T> {
  try {
    return await invoke<T>(cmd, args);
  } catch (e: unknown) {
    // 统一错误处理
    const err = e as { message?: string; [key: string]: unknown };
    if (err.message) {
      throw new Error(err.message);
    }
    throw new Error(String(e));
  }
}
```

### 2.2 Service 层模式

```typescript
// src/services/room.service.ts
import { callCommand } from "./api";
import type { Room, RoomResponse } from "@/types/room";

export const roomService = {
  async list() {
    return callCommand<RoomResponse[]>("list_rooms");
  },

  async get(id: number) {
    return callCommand<RoomResponse>("get_room", { id });
  },

  async update(id: number, data: Partial<Room>) {
    return callCommand<{ success: boolean }>("update_room", { id, ...data });
  },
};
```

## 三、Pinia Store 规范

### 3.1 Store 定义模式

```typescript
// src/stores/room.ts
import { defineStore } from "pinia";
import { ref } from "vue";
import { roomService } from "@/services/room.service";
import type { RoomResponse } from "@/types/room";

export const useRoomStore = defineStore("room", () => {
  const rooms = ref<RoomResponse[]>([]);
  const loading = ref(false);

  async function fetchRooms() {
    loading.value = true;
    try {
      rooms.value = await roomService.list();
    } finally {
      loading.value = false;
    }
  }

  return { rooms, loading, fetchRooms };
});
```

## 四、金额处理

### 4.1 分转元（显示用）

```typescript
// src/utils/money.ts

export function toYuanNumber(fen: number): number {
  return Math.round((fen / 100) * 100) / 100;
}

export function formatMoney(fen: number): string {
  const yuan = toYuanNumber(fen);
  return `¥${yuan.toLocaleString("zh-CN", { minimumFractionDigits: 2 })}`;
}
```

### 4.2 元转分（提交用）

```typescript
export function toCent(yuan: number | string): number {
  const valueStr = typeof yuan === "string" ? yuan : String(yuan);
  const isNegative = valueStr.startsWith("-");
  const absValueStr = isNegative ? valueStr.slice(1) : valueStr;
  const parts = absValueStr.split(".");
  const intPart = parseInt(parts[0] || "0", 10) * 100;
  const decPart = parts[1] ? parts[1].padEnd(2, "0").slice(0, 2) : "00";
  const result = intPart + parseInt(decPart, 10);
  return isNegative ? -result : result;
}
```

```typescript
const NO_CONTRACT_STATUSES = ["空房", "员工", "管理", "违约", "待清洁", "维修中"];

const filteredRooms = computed(() => {
  return rooms.value.filter((room) => {
    // 状态筛选
    if (statusFilter.value && room.status !== statusFilter.value) {
      return false;
    }
    // 关键词搜索

      return false;
    }
    return true;
  });
});
```

## 十、日期处理

### 10.1 日期格式

统一使用 `YYYY-MM-DD` 格式，与后端 SQLite 保持一致。

```typescript
// src/utils/date.ts
export function formatDate(dateStr: string): string {
  if (!dateStr) return "-";
  const date = new Date(dateStr);
  return date.toLocaleDateString("zh-CN", {
    year: "numeric",
    month: "2-digit",
    day: "2-digit",
  });
}
```

### 10.2 当月首末日

```typescript
export function getMonthRange(year: number, month: number) {
  const firstDay = `${year}-${String(month).padStart(2, "0")}-01`;
  const lastDay = new Date(year, month, 0).getDate();
  const lastDayStr = `${year}-${String(month).padStart(2, "0")}-${lastDay}`;
  return { firstDay, lastDayStr };
}
```
